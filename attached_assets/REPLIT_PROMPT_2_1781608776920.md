## ISOMETRIC CANVAS RENDERER

Create `game/renderer.ts`. Use an HTML `<canvas>` element with 2D context. Implement a proper isometric grid renderer:

### Grid coordinate system:
```typescript
// Convert grid [col, row] to isometric screen [x, y]
function isoToScreen(col: number, row: number, tileW: number, tileH: number): { x: number; y: number } {
  return {
    x: (col - row) * (tileW / 2),
    y: (col + row) * (tileH / 2),
  };
}
```

### Tile sizes:
- `TILE_W = 96` (width of one isometric tile in pixels)
- `TILE_H = 48` (height of one isometric tile in pixels)
- Canvas is `responsive` — use `canvas.width = container.clientWidth * devicePixelRatio`

### Animation loop:
```typescript
function startRenderLoop(canvas: HTMLCanvasElement, getState: () => GameState) {
  let lastTime = 0;
  function frame(timestamp: number) {
    const dt = (timestamp - lastTime) / 1000; // seconds
    lastTime = timestamp;
    const state = getState();
    
    const ctx = canvas.getContext('2d')!;
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    // 1. Draw ground tiles
    drawGrid(ctx, state);
    // 2. Draw buildings (sorted by render order: back-to-front, top-to-bottom in iso)
    drawBuildings(ctx, state, timestamp);
    // 3. Draw animated particles (smoke, sparks, money floaties)
    drawParticles(ctx, state.particles, timestamp);
    // 4. Draw UI overlays (resource counts above buildings, selection rings)
    drawBuildingOverlays(ctx, state, timestamp);
    
    requestAnimationFrame(frame);
  }
  requestAnimationFrame(frame);
}
```

## GAME LOOP ENGINE

In `game/engine.ts`:

```typescript
// Tick every second
export function tick(state: GameState, deltaTime: number): GameState {
  const newState = { ...state };
  const dt = deltaTime; // fractional seconds
  
  // 1. Apply HQ global multiplier
  const hqBuilding = state.buildings.find(b => b.id === 'corporate_hq');
  const globalMultiplier = hqBuilding ? (1 + 0.25 * hqBuilding.level) : 1;
  
  // 2. Apply security post crisis damage reduction
  const securityBuildings = state.buildings.filter(b => b.id === 'security_post');
  const crisisDamageReduction = securityBuildings.length * 0.2;
  
  // 3. Apply active crisis modifiers
  const crisisMultipliers = getActiveCrisisMultipliers(state.activeCrisis, crisisDamageReduction);
  
  // 4. Count available workers
  const totalWorkers = 10 + state.buildings.filter(b => b.id === 'worker_camp').length * 5;
  const usedWorkers = state.buildings.reduce((sum, b) => sum + BUILDINGS[b.id].workersRequired, 0);
  const workerUtilization = Math.min(1, totalWorkers > 0 ? 1 - Math.max(0, usedWorkers - totalWorkers) / usedWorkers : 0);
  
  // 5. Process each building
  for (const building of state.buildings) {
    const def = BUILDINGS[building.id];
    if (!def.produces || Object.keys(def.produces).length === 0) continue;
    
    // Check if workers satisfied
    const canOperate = usedWorkers <= totalWorkers;
    if (!canOperate) continue;
    
    // Pump station aura
    const pumpBonus = getPumpStationBonus(building, state.buildings);
    
    // Pipeline efficiency
    const pipelineEfficiency = getPipelineEfficiency(building, state.buildings);
    
    // Level multiplier
    const levelMult = Math.pow(def.productionPerLevel, building.level - 1);
    
    const effMult = globalMultiplier * pumpBonus * pipelineEfficiency * levelMult;
    
    // Produce
    for (const [resId, baseRate] of Object.entries(def.produces)) {
      const crisisMult = crisisMultipliers[building.id]?.[resId] ?? 1;
      const produced = baseRate * effMult * crisisMult * dt;
      const cap = getStorageCap(resId as ResourceId, state);
      newState.resources[resId as ResourceId] = Math.min(
        (newState.resources[resId as ResourceId] ?? 0) + produced,
        cap
      );
    }
    
    // Consume
    for (const [resId, baseRate] of Object.entries(def.consumes)) {
      const consumed = baseRate * levelMult * dt;
      const available = newState.resources[resId as ResourceId] ?? 0;
      if (available < consumed) {
        // Stall the building if not enough input resource
        // Remove what was just produced
        // (simplified: just don't produce, handled by checking ahead)
      }
      newState.resources[resId as ResourceId] = Math.max(0, available - consumed);
    }
  }
  
  // 6. Process crisis event countdown
  if (state.activeCrisis) {
    newState.crisisTimeLeft = state.crisisTimeLeft - dt;
    if (newState.crisisTimeLeft <= 0) {
      // Resolve crisis — give reward
      const crisis = CRISIS_EVENTS.find(e => e.id === state.activeCrisis)!;
      for (const [res, amount] of Object.entries(crisis.rewardOnResolve)) {
        newState.resources[res as ResourceId] = (newState.resources[res as ResourceId] ?? 0) + amount;
      }
      newState.activeCrisis = null;
      newState.crisisTimeLeft = 0;
      newState.notifications.push({ id: Date.now(), text: `✅ Кризис "${crisis.name}" разрешён!`, type: 'success' });
    }
  }
  
  // 7. Check for new crisis trigger (every 5 real minutes, random)
  if (!state.activeCrisis && Math.random() < 0.0003 * dt) { // ~once per ~55min on average
    const eligible = CRISIS_EVENTS.filter(e => 
      e.minPlayerLevel <= state.level && e.triggerChance > Math.random()
    );
    if (eligible.length > 0) {
      const crisis = eligible[Math.floor(Math.random() * eligible.length)];
      newState.activeCrisis = crisis.id;
      newState.crisisTimeLeft = crisis.duration;
      newState.notifications.push({ id: Date.now(), text: `⚠️ ${crisis.name}`, type: 'crisis' });
    }
  }
  
  // 8. Daily login reward (check date)
  // (implement separately in UI)
  
  // 9. Auto-save every 30 seconds
  if (Date.now() - state.lastSaved > 30000) {
    saveGame(newState);
    newState.lastSaved = Date.now();
  }
  
  return newState;
}
```

## PARTICLES SYSTEM

Add a lightweight particle system for visual juice:

```typescript
interface Particle {
  id: number;
  x: number; y: number;
  vx: number; vy: number;
  life: number;     // 0-1
  type: 'coin' | 'smoke' | 'spark' | 'xp' | 'oil_drop';
  color: string;
  size: number;
}

// Spawn particles when a building produces:
// - Coin particle: floats up from AZS, shows "+" and a 💰
// - Smoke: drifts up from refinery/hq
// - Oil drop: flows along pipeline
// - XP: green stars from lab/buildings on level up
// - Spark: orange sparks from crisis events

function updateParticles(particles: Particle[], dt: number): Particle[] {
  return particles
    .map(p => ({ ...p, x: p.x + p.vx * dt, y: p.y + p.vy * dt, life: p.life - dt * 0.5 }))
    .filter(p => p.life > 0);
}

function drawParticles(ctx: CanvasRenderingContext2D, particles: Particle[], timestamp: number) {
  for (const p of particles) {
    ctx.save();
    ctx.globalAlpha = p.life;
    ctx.fillStyle = p.color;
    ctx.font = `${p.size}px system-ui`;
    ctx.fillText(
      p.type === 'coin' ? '💰' : p.type === 'xp' ? '⭐' : p.type === 'smoke' ? '💨' : '✨',
      p.x, p.y
    );
    ctx.restore();
  }
}
```
