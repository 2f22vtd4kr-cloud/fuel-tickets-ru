  haptic.error();                    // ← Strong error haptic
  console.error(`Error ${context}:`, message);
  
  // Show user-friendly toast
  showToast({
    type: 'error',
    title: 'Ошибка',
    message: message || 'Что-то пошло не так',
    duration: 4000
  });
};
